0Setup — Build the Videogames Database
Today's Dataset

Fresh dataset — a videogames database with two linked tables: games (10 titles) and studios (5 developers). The tables are already linked with a foreign key so you can practice JOINs right away.

📦 Setup — Copy, Paste, Run in Supabase SQL Editor
-- Drop old tables if they exist
DROP TABLE IF EXISTS games;
DROP TABLE IF EXISTS studios;

-- Create studios table
CREATE TABLE studios (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  country TEXT,
  founded INT4
);

INSERT INTO studios (name, country, founded) VALUES
  ('Nintendo', 'Japan', 1889),
  ('Rockstar Games', 'USA', 1998),
  ('CD Projekt Red', 'Poland', 1994),
  ('FromSoftware', 'Japan', 1986),
  ('Mojang', 'Sweden', 2009);

-- Create games table with foreign key
CREATE TABLE games (
  id SERIAL PRIMARY KEY,
  title TEXT NOT NULL,
  genre TEXT,
  release_year INT4,
  rating FLOAT4,
  copies_sold_millions FLOAT4,
  studio_id INT4
);

INSERT INTO games (title, genre, release_year, rating, copies_sold_millions, studio_id) VALUES
  ('Breath of the Wild', 'Adventure', 2017, 9.7, 31.2, 1),
  ('Mario Kart 8', 'Racing', 2014, 9.2, 64.3, 1),
  ('GTA V', 'Action', 2013, 9.5, 200.0, 2),
  ('Red Dead Redemption 2', 'Action', 2018, 9.7, 57.0, 2),
  ('The Witcher 3', 'RPG', 2015, 9.3, 50.0, 3),
  ('Cyberpunk 2077', 'RPG', 2020, 7.2, 25.0, 3),
  ('Elden Ring', 'RPG', 2022, 9.6, 23.0, 4),
  ('Dark Souls III', 'RPG', 2016, 8.9, 10.0, 4),
  ('Minecraft', 'Sandbox', 2011, 9.0, 300.0, 5),
  ('Minecraft Dungeons', 'Action', 2020, 6.8, 15.0, 5);
